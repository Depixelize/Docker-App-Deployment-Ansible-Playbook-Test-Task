---
- name: Install Apache Docker Container with PHP 7.2 and Nginx with GeoIP support
  hosts: localhost
  become: true
  tasks:
    - name: Install Docker
      apt:
        name: docker.io
        state: present
      register: docker_install

    - name: Install Python Pip
      apt:
        name: python3-pip
        state: present
      register: python_pip_install

    - name: Install Docker-py
      pip:
        name: docker-py
        state: present

    - name: Pull Apache Docker Image
      docker_image:
        name: httpd:latest
        source: pull
        state: present

    - name: Pull PHP 7.2 Docker Image
      docker_image:
        name: php:7.2-apache
        source: pull
        state: present

    - name: Pull Nginx Docker Image
      docker_image:
        name: nginx:latest
        source: pull
        state: present

    - name: Create custom Docker network
      command: docker network create --subnet=10.0.0.0/24 custom_network

    - name: Create Apache Container
      docker_container:
        name: apache
        image: httpd:latest
        state: started
        ports:
          - "8080:8080"
        networks:
          - name: custom_network
            ipv4_address: 10.0.0.2

    - name: Create PHP 7.2 Container
      docker_container:
        name: php
        image: php:7.2-apache
        state: started
        ports:
          - "9001:9001"
        networks:
          - name: custom_network
            ipv4_address: 10.0.0.3

    - name: Create Nginx Container
      docker_container:
        name: nginx
        image: nginx:latest
        state: started
        ports:
          - "80:80"
        networks:
          - name: custom_network
            ipv4_address: 10.0.0.4

    - name: Download GeoIP.dat.gz
      get_url:
        url: https://mailfud.org/geoip-legacy/GeoIP.dat.gz
        dest: /tmp/GeoIP.dat.gz

    - name: Unpack GeoIP.dat.gz
      command: gunzip /tmp/GeoIP.dat.gz

    - name: Copy GeoIP.dat to Nginx Container
      docker_container:
        name: nginx
        state: started
        volumes:
          - /tmp/GeoIP.dat:/etc/nginx/GeoIP.dat
